# Pipelines de modelos

En esta sección ilustraremos cómo estructurar nuestros proyectos para migrarlos fácilmente a un entorno de producción. Esto significa, adaptarnos a escribir código de manera que sea lo más fácil posible de distribuirlo e integrarlo con otros servicios. Vamos a ilustrar el procedimiento con unos datos reales.

# Escribiendo la transformación como una clase

Vamos a convertir esta sencilla transformación en una clase, con métodos fit_transform, transform y __repr__.

from sklearn.preprocessing import QuantileTransformer
import pandas as pd
import random
import string

def generate_id(n=4, chars=string.hexdigits):
    return ''.join(random.choice(chars) for _ in range(n))

class MyPipe:
    
    def __init__(self):
        self.id = generate_id()
        self.amt_transformer = None
    
    def fit_transform(self, data):
        self.amt_transformer = QuantileTransformer(n_quantiles=50)
        data.loc[:,'Q'] = self.amt_transformer.fit_transform(data[['CreditAmount']])
        return data[['Q', 'CreditAmount']]
    
    def transform(self, data):
        data.loc[:,'Q'] = self.amt_transformer.transform(data[['CreditAmount']])
        return data[['Q', 'CreditAmount']]

    def __repr__(self):
        return f'MyPipe object with id {self.id}'

# Añadiendo un modelo

Ahora vamos a añadir el modelo que utiliza la columna Q para clasificar los créditos.

from joblib import dump, load

class MyModel:
    
    def __init__(self):
        self.id = generate_id()
        self.pipe = MyPipe()
        self.rules = {'low': 0.5, 'high': 0.95, 'very-high': 1.1}
        
    def fit(self, data):
        data = self.pipe.fit_transform(data)
    
    def predict(self, data):
        data = self.pipe.transform(data)
        pred = ['']*len(data)
        for i in range(len(data)):
            q = data.Q.values[i]
            for k in self.rules.keys():
                if self.rules[k] > q:
                    pred[i] = k
                    break
        return pd.DataFrame({'prediction': pred, 'Q': data.Q, 'CreditAmount': data.CreditAmount})
    
    def __repr__(self):
        return f'MyModel with id {self.id}'
    
    def save(self, path):
        dump(self, f'{path}/{self.id}')
    
    def load(model_id, path):
        obj = load(f'{path}/{model_id}')
        return obj

model = MyModel()

# Ejercicio

A la entidad bancaria le interesa asignar a cada cliente un clúster basado en la edad y la duración y la magnitud del crédito, de forma que obtenga los siguientes grupos:

  • Long-term credits, young people
  • Short-term credits, young people
  • Short-term credits, mature people
  • Long-term credits, small amounts, mature people
  • Long-term credits, large amounts, mature people

Debes crear una estructura de modelo, de forma que, dado un diccionario con keys 'DurationMonths', 'Age', 'CreditAmount', el modelo retorne el clúster que corresponde.
