# Visualización de datos en python con plotly.js

En esta sección estudiaremos algunas funciones básicas para visualización del paquete plotly.

import plotly.express as px
import pandas as pd

Utilizaremos el dataset de créditos bancarios de ejemplos anteriores.

df = pd.read_csv('data/german_train.csv')

# Scatter Plots

px.scatter(df, x='DurationMonth', y='CreditAmount')

# Line Charts

tmp = df.groupby('Age')[['CreditAmount']].max()
tmp['Age'] = tmp.index
px.line(tmp, x='Age', y='CreditAmount')

# Bar Charts

tmp = df.groupby('StatusAccount')[['StatusAccount']].count()
tmp['count'] = tmp['StatusAccount']
tmp['StatusAccount'] = tmp.index
px.bar(tmp, x='StatusAccount', y='count')


px.bar(df.sample(100), x='StatusAccount', y='CreditAmount', color='StatusAndSex')

# Pie Charts

tmp = df.groupby('StatusAccount')[['StatusAccount']].count()
tmp['count'] = tmp['StatusAccount']
tmp['StatusAccount'] = tmp.index

px.pie(tmp, values='count', names='StatusAccount')


px.pie(tmp, values='count', names='StatusAccount', hole=.5)

# Histograms

px.histogram(df, 'CreditAmount', nbins=20)


px.histogram(df, x='CreditAmount', color='StatusAndSex', nbins=20)

# 3D Scatters

px.scatter_3d(df.sample(100), x='DurationMonth', y='Age', z='CreditAmount', color='Job')
